            <script>
                function loadDoc() {
                    const xhttp = new XMLHttpRequest();
                    xhttp.onload = function() {
                    document.getElementById("demo").innerHTML =
                    this.responseText;
                    }
                    xhttp.open("GET", "ajax_info.txt");
                    xhttp.send();
                }
            </script>
            <script src="https://cdn.jsdelivr.net/npm/axios/dist/axios.min.js"></script>
            <script>
                const getResponse = document.getElementById('getResponse');


                axios.get('https://jsonplaceholder.typicode.com/posts', {
                    params: {
                        userId: 1 // Параметры запроса передаются через объект params
                    }
                })
                    .then(response => {

                        for (let i = 0; i < response.data.length; i++) {
                            const itemObject = response.data[i];
                            let ulContainer = document.createElement("ul");
                            getResponse.appendChild(ulContainer);
                            for (const item in itemObject) {
                                let liContainer = document.createElement("li");
                                ulContainer.appendChild(liContainer)
                                    .textContent = `${item}: ${itemObject[item]}`;
                            }
                        }
                        console.log(response.data);
                    })
                    .catch(error => {
                        getResponse.textContent = `ERROR!: ${error}`;
                        console.error('Ошибка при выполнении запроса: ', error);
                    });
            </script>
